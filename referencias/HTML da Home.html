Script pronto para colar no site
<script>
var TOKEN = '41c655ef2675ac216bb47be6d40cbd81871d5595842f7eae3864d05b0f7b0419';
(function(){
  var API = 'https://track.pagah.com/api/track/site/' + encodeURIComponent(TOKEN);
    function uuid4(){return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g,function(c){var r=Math.random()*16|0,v=c==='x'?r:(r&0x3|0x8);return v.toString(16)})}
    function getCookie(name){var m=document.cookie.match('(^|;)\\s*'+name+'\\s*=\\s*([^;]+)');return m?m.pop():null}
    function setCookie(name,value,days){var d=new Date();d.setTime(d.getTime()+days*24*60*60*1000);document.cookie=name+'='+value+'; expires='+d.toUTCString()+'; path=/; SameSite=Lax'}
    function getStableSessionId(){
        var sid = getCookie('pagah_tid') || localStorage.getItem('pagah_tid');
        if (!sid) {
            sid = uuid4();
            localStorage.setItem('pagah_tid', sid);
            setCookie('pagah_tid', sid, 365);
        }
        return sid;
    }
  function send(){
    var payload = {
            session_id: getStableSessionId(),
      type: 'pageview',
    path: window.location.pathname || '/track-32',
      query: {},
      timestamp: Math.floor(Date.now()/1000)
    };
    fetch(API, {
      method: 'POST',
      headers: {'Content-Type':'application/json'},
      body: JSON.stringify(payload)
    }).then(function(r){ console.log('status', r.status); }).catch(function(e){ console.error(e); });
  }
  send();
})();
</script>
